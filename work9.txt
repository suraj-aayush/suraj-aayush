		Q9. Write a structure to store the names, salary and working hours per day (fixed at the time 
		of joining) of 10 employees. Calculate the increased monthly salary depending on the 
		working hours per day as given below. Print the name of all the employees along with their 
		new salaries.
		Hours of work per day:   8         10          >=12
		Increase in salary   :   Rs.2500   Rs. 5000    Rs.7500      */


		#include<stdio.h>         
		typedef struct
		{
		    char str[20];
		    int sl,whour;
		}student;
		int main()
		{
			student s[10];
			int i;
			printf("\t\t\t******INPUT******\n");
			for(i=0;i<10;i++)
			{
			    printf("enter name,salary and working hour per day:\n");
			    scanf("\n%[^\n]%*c",s[i].str);
			    scanf("%d%d",&s[i].sl,&s[i].whour);           //working hour 8,10 or >=12 only.
		
			}
			printf("\t\t\t******OUTPUT******\n");
			for(i=0;i<10;i++)
			{
			    if(s[i].whour==8)
			    {
			        printf("Name:%s",s[i].str);
			        printf("\tNew salaries:%d\n",s[i].sl+2500);
			    }
			    if(s[i].whour==10)
			    {
			        printf("Name:%s\n",s[i].str);
			        printf("\tNew salaries:%d\n",s[i].sl+5000);
			    }
			    if(s[i].whour>=12)
			    {
			        printf("Name:%s\n",s[i].str);
			        printf("\tNew salaries:%d\n",s[i].sl+7500);
			    }
			}
			return 0;
		}
